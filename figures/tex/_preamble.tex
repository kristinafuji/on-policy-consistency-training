% figure. Keeping this in one place keeps the per-figure files short and

% ---- Base axis styling --------------------------------------------------
\pgfplotsset{
  every axis/.append style={
    axis line style={axisEdge, line width=0.6pt},
    tick style={axisEdge, line width=0.6pt},
    tick label style={font=\small},
    label style={font=\small},
    title style={font=\small\bfseries, yshift=-1pt},
    legend style={
      font=\small,
      draw=none,
      fill=none,
      cells={anchor=west},
      /tikz/every even column/.append style={column sep=4pt},
    },
    axis on top,
    axis x line*=bottom,
    axis y line*=left,
  },
}

% TODO overlay — used when an entire panel has no data.
\newcommand{\TODOoverlay}[1][]{%
  \node[#1, anchor=center, font=\bfseries\large, text=todoGray]
    at (rel axis cs:0.5,0.5) {TODO};
}

% Unified bar style applied via `every axis/.append style` so it survives
% the groupplot's option-parser evaluation order (pgfplots won't accept
% `bar width=...` in the groupplot option list itself — `ybar` must be
% activated first, and that only happens per-axis).
\pgfplotsset{
  barPanel/.style={
    ybar,
    bar width=10pt,
  },
}
